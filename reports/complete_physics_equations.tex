\subsection{1.1 Rotational Dynamics (Euler's Equations)}

\begin{longtable}{|p{0.2\textwidth}|p{0.35\textwidth}|p{0.35\textwidth}|}
\hline
\textbf{Angular Acceleration} & \textbf{Equation} & \textbf{Used In (File:Line)} \\
\hline
Roll ($\dot{p}$) &
$\dot{p} = \frac{J_{yy} - J_{zz}}{J_{xx}} \cdot q \cdot r + \frac{\tau_x}{J_{xx}}$ &
\textbf{Data Gen:} \texttt{generate\_quadrotor\_data.py:243} \newline
\textbf{PINN Loss:} \texttt{pinn\_model\_optimized\_v2.py:108} \\
\hline
Pitch ($\dot{q}$) &
$\dot{q} = \frac{J_{zz} - J_{xx}}{J_{yy}} \cdot p \cdot r + \frac{\tau_y}{J_{yy}}$ &
\textbf{Data Gen:} \texttt{generate\_quadrotor\_data.py:244} \newline
\textbf{PINN Loss:} \texttt{pinn\_model\_optimized\_v2.py:109} \\
\hline
Yaw ($\dot{r}$) &
$\dot{r} = \frac{J_{xx} - J_{yy}}{J_{zz}} \cdot p \cdot q + \frac{\tau_z}{J_{zz}}$ &

\textbf{PINN Loss:} \texttt{pinn\_model\_optimized\_v2.py:110} \\
\hline
\end{longtable}

\textbf{Shorthand notation:} $t_1 = \frac{J_{yy} - J_{zz}}{J_{xx}}$, $t_2 = \frac{J_{zz} - J_{xx}}{J_{yy}}$, $t_3 = \frac{J_{xx} - J_{yy}}{J_{zz}}$

\textbf{Key Feature:} \textcolor{red}{\textbf{NO artificial damping terms}} (removed $-2p$, $-2q$, $-2r$). Pure Euler equations.

\textbf{PINN Loss:} \texttt{pinn\_model\_optimized\_v2.py:113} \\

\textbf{PINN Loss:} \texttt{pinn\_model\_optimized\_v2.py:115} \\
\hline
\end{longtable}

\textbf{Singularity:} Gimbal lock occurs at $\theta = \pm 90°$ (vertical climb/dive).

\textbf{Angle Wrapping:} $\phi, \theta, \psi \in [-\pi, \pi]$ using \texttt{arctan2(sin($\cdot$), cos($\cdot$))}

(Full 3D dynamics in data generation) \\
\hline
Z-axis acceleration ($\dot{w}$) &
$\dot{w} = q \cdot u - p \cdot v + \frac{f_z}{m} + g \cos(\theta) \cos(\phi) - 0.05 \cdot w \cdot |w|$ &

\textbf{PINN (simplified):} \texttt{pinn\_model\_optimized\_v2.py:119} \\

\textbf{Forces:} $f_x = 0$, $f_y = 0$, $f_z = -T$ (thrust acts only in body z-direction)

\textbf{Key Feature:} \textcolor{red}{\textbf{Quadratic aerodynamic drag}} $F_d = 0.05 \cdot v \cdot |v|$ (NOT linear)

\textbf{Simplified for PINN (Vertical Only):}

\textbf{Source:} \texttt{pinn\_model\_optimized\_v2.py:119}

$$\dot{w} = -\frac{T \cos(\theta) \cos(\phi)}{m} + g - 0.05 \cdot v_z \cdot |v_z|$$

where $v_z = w$ is vertical velocity in NED frame.

\textbf{Simplified for PINN (Vertical Only):} $\dot{z} = v_z$

\section{Motor \& Actuator Dynamics}

\subsection{2.1 First-Order Motor Lag (Realistic Actuator Response)}

\textbf{Source:} \texttt{generate\_quadrotor\_data.py:118-133}

\begin{equation}
y_{actual}(t + \Delta t) = y_{actual}(t) + \alpha \cdot [y_{cmd}(t) - y_{actual}(t)]
\end{equation}

where $\alpha = \frac{\Delta t}{\tau + \Delta t}$ and $\tau = 0.08$ s (80ms time constant)

\textbf{Applied to:} Thrust ($T$), Roll torque ($\tau_x$), Pitch torque ($\tau_y$), Yaw torque ($\tau_z$)

\textbf{Physical Meaning:} Motors cannot respond instantaneously; they have spin-up/spin-down lag.

\subsection{2.2 Slew Rate Limiting}

\textbf{Source:} \texttt{generate\_quadrotor\_data.py:98-116}

\begin{equation}
y_{limited}(t + \Delta t) =
\begin{cases}
y_{cmd}(t) & \text{if } |y_{cmd}(t) - y_{current}(t)| \leq R_{max} \cdot \Delta t \\
y_{current}(t) + \text{sign}(y_{cmd} - y_{current}) \cdot R_{max} \cdot \Delta t & \text{otherwise}
\end{cases}
\end{equation}

\textbf{Rates:}
\begin{itemize}
\item Thrust slew rate: $R_{T,max} = 15.0$ N/s
\item Torque slew rate: $R_{\tau,max} = 0.5$ N·m/s
\end{itemize}

\textbf{Physical Meaning:} Limits maximum rate of change to prevent unrealistic step changes.

\section{Control \& Filtering}

\subsection{3.1 Low-Pass Filter (Reference Smoothing)}

\textbf{Source:} \texttt{generate\_quadrotor\_data.py:81-96}

\begin{equation}
r_{filtered}(t + \Delta t) = r_{filtered}(t) + \alpha \cdot [r_{raw}(t) - r_{filtered}(t)]
\end{equation}

where $\alpha = \frac{\Delta t}{\tau_{LPF} + \Delta t}$ and $\tau_{LPF} = 0.4$ s (400ms time constant)

\textbf{Purpose:} Smooth square wave setpoint transitions to prevent high-frequency oscillations.

\subsection{3.2 Cascaded PID Controllers}

\textbf{Source:} \texttt{generate\_quadrotor\_data.py:196-224}

\subsubsection{3.2.1 Roll Controller}

\textbf{Outer Loop (Angle):}
\begin{equation}
p_{ref} = k_1 \cdot (\phi_{ref} - \phi) + k_i \cdot \int (\phi_{ref} - \phi) \, dt
\end{equation}

\textbf{Inner Loop (Rate):}
\begin{equation}
\tau_x = k_2 \cdot (p_{ref} - p)
\end{equation}

\textbf{Gains:} $k_1 = 0.8$, $k_i = 0.002$, $k_2 = 0.05$

\subsubsection{3.2.2 Pitch Controller}

\textbf{Outer Loop (Angle):}
\begin{equation}
q_{ref} = k_{11} \cdot (\theta_{ref} - \theta) + k_{i1} \cdot \int (\theta_{ref} - \theta) \, dt
\end{equation}

\textbf{Inner Loop (Rate):}
\begin{equation}
\tau_y = k_{21} \cdot (q_{ref} - q)
\end{equation}

\textbf{Gains:} $k_{11} = 0.8$, $k_{i1} = 0.002$, $k_{21} = 0.05$

\subsubsection{3.2.3 Yaw Controller}

\textbf{Outer Loop (Angle):}
\begin{equation}
r_{ref} = k_{12} \cdot (\psi_{ref} - \psi) + k_{i2} \cdot \int (\psi_{ref} - \psi) \, dt
\end{equation}

\textbf{Inner Loop (Rate):}
\begin{equation}
\tau_z = k_{22} \cdot (r_{ref} - r)
\end{equation}

\textbf{Gains:} $k_{12} = 0.8$, $k_{i2} = 0.002$, $k_{22} = 0.05$

\subsubsection{3.2.4 Altitude Controller}

\textbf{Outer Loop (Position):}
\begin{equation}
v_{z,ref} = k_{z1} \cdot (z_{ref} - z) + k_{z2} \cdot \int (z_{ref} - z) \, dt
\end{equation}

\textbf{Inner Loop (Velocity):}
\begin{equation}
T = k_v \cdot (v_{z,ref} - w)
\end{equation}

\textbf{Gains:} $k_{z1} = 1.5$, $k_{z2} = 0.15$, $k_v = -0.25$ (negative for NED frame)

\subsection{3.3 Square Wave Reference Generation}

\textbf{Source:} \texttt{generate\_quadrotor\_data.py:73-79}

\begin{equation}
r_{square}(t) =
\begin{cases}
A_{low} & \text{if } (t \bmod T) / T < 0.5 \\
A_{high} & \text{otherwise}
\end{cases}
\end{equation}

where $T$ is period, $A_{low}$ and $A_{high}$ are amplitudes.

\section{Energy \& Conservation}

\subsection{4.1 Kinetic Energy}

\textbf{Source:} \texttt{pinn\_model\_optimized\_v2.py:204-205}

\begin{equation}
KE = \frac{1}{2} m v_z^2 + \frac{1}{2} (J_{xx} p^2 + J_{yy} q^2 + J_{zz} r^2)
\end{equation}

\textbf{Components:}
\begin{itemize}
\item Translational: $\frac{1}{2} m v_z^2$ (vertical velocity only in PINN)
\item Rotational: $\frac{1}{2} (J_{xx} p^2 + J_{yy} q^2 + J_{zz} r^2)$
\end{itemize}

\subsection{4.2 Potential Energy}

\textbf{Source:} \texttt{pinn\_model\_optimized\_v2.py:208-209}

\begin{equation}
PE = m \cdot g \cdot z
\end{equation}

where $z < 0$ for altitude above ground in NED frame.

\subsection{4.3 Energy Consistency Loss}

\textbf{Source:} \texttt{pinn\_model\_optimized\_v2.py:212}

\begin{equation}
\mathcal{L}_{energy} = \mathbb{E}[(E_{t+1} - E_t)^2]
\end{equation}

where $E = KE + PE$

\textbf{Purpose:} Penalize unphysical energy changes between timesteps.
